# GitHub Push Commands
# Run these commands AFTER creating your repository on GitHub

# 1. Set your Git identity (if not already set)
git config --global user.name "Your Name"
git config --global user.email "your-email@example.com"

# 2. Rename branch to 'main' (modern convention)
git branch -M main

# 3. Add your GitHub repository as remote
# Replace YOUR-USERNAME and YOUR-REPO-NAME with your actual values
git remote add origin https://github.com/YOUR-USERNAME/YOUR-REPO-NAME.git

# 4. Push to GitHub
git push -u origin main

# Example:
# git remote add origin https://github.com/johndoe/BiketaBai3.0.git
# git push -u origin main
